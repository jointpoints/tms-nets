35  template<uint32_t P>
36  polynomialgf<P> gcd(polynomialgf<P> m, polynomialgf<P> n) {
37  if (m.is_zero() || n.is_zero()) {
38  throw ::std::domain_error("arguments must be strictly positive");
39  }
40  if (m.degree() < n.degree()) {
41  ::std::swap(m, n);
42  }
43  polynomialgf<P> u0 = m, u1 = polynomialgf<P>({1}), u2 = polynomialgf<P>({0}),
44  v0 = n, v1 = polynomialgf<P>({0}), v2 = polynomialgf<P>({1}),
45  w0, w1, w2, q;
46  while (!v0.is_zero()) {
47  q = u0 / v0;
48  w0 = u0 - q * v0, w1 = u1 - q * v1, w2 = u2 - q * v2;
49  u0 = v0, u1 = v1, u2 = v2, v0 = w0, v1 = w1, v2 = w2;
50  }
51  return u0;
52  }

83  template<uint32_t P>
84  [[nodiscard]]
85  polynomialgf<P> x_pow_mod_sub(
86  uint64_t pow,
87  const polynomialgf<P> &mod,
88  const polynomialgf<P> &sub = polynomialgf<P>({0})
89  ) noexcept {
90  // Эта функция возврадает по сути следующий редультат:
91  // return ((polynomialgf<P>({1}) << pow) - sub) % mod;
92  // её существование необходимо, т.к. в методе Рабина
93  // и в методе проверки на примитивность требуется вычислить
94  // её результат для столь больших pow, что x^pow не влезает
95  // в оперативную память, данный метод решает эту проблему,
96  // но он всё ещё адски медленный
97  const auto n = mod.degree();
98  polynomialgf<P> xn = polynomialgf<P>{1} << n; // x^n
99  polynomialgf<P> res({1});
100 
101  uint64_t d = 0;
102  typename polynomialgf<P>::size_type tmp;
103  for (auto m = res.degree(); pow + m >= n; m = res.degree()) {
104  tmp = n - m;
105  pow -= tmp;
106  res <<= tmp;
107  if (res == xn) {
108  // деление в столбик начинается с деления x^n
109  if (d == 0) { d = pow; }
110  // при больших pow деление успевает зациклиться и мы снова придём к x^n
111  else {
112  // находим потерю в степени, произошедшую до этого момента
113  d -= pow;
114  // можем пропустить кучу бесполезных шагов, повторяющих уже сделанные
115  pow %= d;
116  // после этой операции в этот if мы больше не попадём,
117  // т.к. pow < d и деление в столбик не дойдёт до x^n снова,
118  // поэтому d можно не обнулять
119  }
120  }
121  res %= mod;
122  }
123  return (res << pow) - sub;
124  }

160  // функция для построения матрицы берлекампа и вычисления её ранга
161  auto berlekampMatrixRank = [](const polynomialgf<P> &val) {
162  polynomialgf<P> tmp;
163  typename polynomialgf<P>::size_type i, j, k, l;
164  const gf<P> zer = 0;
165  const auto n = val.degree();
166  ::std::vector<::std::vector<gf<P>>> m(n, ::std::vector<gf<P>>(n, zer)); // M = 0
167  for (i = 0; i < n; ++i) {
168  tmp = detail::x_pow_mod_sub(i * P, val); // M[i,*] = x ^ ip (mod val)
169  for (j = 0, k = tmp.degree(); j <= k; ++j) {
170  m[i][j] += tmp[j];
171  }
172  m[i][i] -= 1; // M - E
173  }
174 
175  // приведение матрицы к ступенчатому виду
176  bool f;
177  gf<P> mul;
178  for (i = k = 0; i < n && k < n; ++k) {
179  f = !m[i][k].is_zero();
180  for (j = i + 1; j < n; ++j) {
181  if (!m[j][k].is_zero()) {
182  if (f) {
183  mul = m[i][k].mul_inv() * m[j][k];
184  m[j][k] = zer;
185  for (l = k + 1; l < n; ++l) {
186  m[j][l] -= m[i][l] * mul;
187  }
188  } else {
189  for (l = k; l < n; ++l) {
190  ::std::swap(m[i][l], m[j][l]);
191  }
192  f = true;
193  }
194  }
195  }
196  i += f;
197  }
198  return i;
199  };

202  auto d = detail::derivative(val);
203  return !d.is_zero() && gcd(val, d).degree() == 0 &&
204  berlekampMatrixRank(val) == val.degree() - 1;

236 
237  // проверка вырожденных случаев
238  if (n == 0 || (val[0].is_zero() && n > 1)) { return false; }

241  // выполняется нормировка, т.к. данный алгоритм справедлив только
242  // для многочленов со старшим коэффициентом, равным единице
243  // умножение многочлена на число не меняет его примитивность
244  const auto poly = val / val[n];
245 
246  // ещё один вырожденный случай, на работу с которым алгоритм не рассчитан
247  if (P == 2 && poly == polynomialgf<P>({1, 1})) { return false; }

249  auto mp = (n % 2) ? -poly[0] : poly[0];

251  // функция для разложения (факторизации) числа на множители
252  // единица и само число (в случае его простоты) в разложение не входят
253  auto factorize = [](uint64_t n) {
254  ::std::vector<uint64_t> list;
255  const auto begin = n;
256  for (uint64_t d = 2; d * d <= n; ++d) {
257  if (n % d) { continue; }
258  list.emplace_back(d);
259  while (n % d == 0) { n /= d; }
260  }
261  if (n != 1 && n != begin) { list.emplace_back(n); }
262  return list;
263  };
264 
265  // проверяется выполнение первого условия
266  if (P > 2) {
267  const auto p = P - 1;
268  auto list = (p == 2) ? ::std::vector<uint64_t>{2} : factorize(p);
269  auto m = list.size() - 1;
270  auto tmp = mp;
271  for (uint32_t i = 1; i <= p; ++i, tmp *= mp) {
272  if (i != p / list[m]) { continue; }
273  if (tmp.data() == 1) { return false; }
274  if (m == 0) { break; } else { m -= 1; }
275  }
276  }

278  // проверяется выполнение второго условия
279  uint64_t r = (detail::integer_power(static_cast<uint64_t>(P), n) - 1) / (P - 1);
280  auto tmp = detail::x_pow_mod_sub(r, val, polynomialgf<P>({mp}));
281  if (!tmp.is_zero()) { return false; }
282
